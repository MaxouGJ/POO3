public static void anniversaires(Animal[] zoo){
	for(int i=0; i < zoo.length; i++){
		zoo[i].age++;
		if(zoo[i].estMort()){
			zoo[i] = null;
		}	
	}
	for(int i=0; i < zoo.length-1; i++){
		if(zoo[i] == null)
			zoo[i] = zoo[i+1];
	}
}
